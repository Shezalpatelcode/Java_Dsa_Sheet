import java.util.*;

class Solution {
    public String findLargest(int[] arr) {
        // 1. Convert int array to String array
        String[] strings = new String[arr.length];
        for (int i = 0; i < arr.length; i++) {
            strings[i] = String.valueOf(arr[i]);
        }

        // 2. Sort strings using a custom comparator
        // We compare (s2 + s1) to (s1 + s2) to sort in descending order of value
        Arrays.sort(strings, new Comparator<String>() {
            @Override
            public int compare(String s1, String s2) {
                String order1 = s1 + s2;
                String order2 = s2 + s1;
                return order2.compareTo(order1); 
            }
        });

        // 3. Handle the edge case where the largest number is 0
        // (e.g., input is [0, 0]), we should return "0" instead of "00"
        if (strings[0].equals("0")) {
            return "0";
        }

        // 4. Build the final string
        StringBuilder sb = new StringBuilder();
        for (String s : strings) {
            sb.append(s);
        }

        return sb.toString();
    }
}

class Solution {
    public String convert(String s, int numRows) {
        // Edge case: If only 1 row or string is too short, no zigzag happens
        if (numRows == 1 || s.length() <= numRows) {
            return s;
        }

        // 1. Create a list of StringBuilders (one for each row)
        StringBuilder[] rows = new StringBuilder[numRows];
        for (int i = 0; i < numRows; i++) {
            rows[i] = new StringBuilder();
        }

        int currentRow = 0;
        boolean goingDown = false;

        // 2. Iterate through the string
        for (char c : s.toCharArray()) {
            rows[currentRow].append(c);

            // 3. Flip direction if we hit the top or bottom row
            if (currentRow == 0 || currentRow == numRows - 1) {
                goingDown = !goingDown;
            }

            // 4. Move to the next row based on direction
            currentRow += goingDown ? 1 : -1;
        }

        // 5. Combine all rows into one final string
        StringBuilder result = new StringBuilder();
        for (StringBuilder row : rows) {
            result.append(row);
        }

        return result.toString();

class Solution {
    public String intToRoman(int num) {
        // Define all Roman symbols, including subtractive combinations, in descending order
        int[] values = {1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1};
        String[] symbols = {"M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I"};
        
        StringBuilder sb = new StringBuilder();
        
        // Loop through the values from largest to smallest
        for (int i = 0; i < values.length; i++) {
            // While the current value can be subtracted from num, add the symbol
            while (num >= values[i]) {
                num -= values[i];
                sb.append(symbols[i]);
            }
        }
        
        return sb.toString();
    }
}

function collision(a, b) {
    let r1 = a.getBoundingClientRect();
    let r2 = b.getBoundingClientRect();
    return !(r1.bottom < r2.top || r1.top > r2.bottom || r1.right < r2.left || r1.left > r2.right);
}

function gameLoop() {
    if (gameOver) return;

    // road move
    road.style.top = (parseInt(road.style.top || -600) + speed) + "px";
    if (parseInt(road.style.top) >= 0) road.style.top = "-600px";

    // enemies
    enemies.forEach(e => {
        let top = parseInt(e.style.top);
        e.style.top = top + speed + "px";

            resetEnemy(e);

    if (coinTop > 600) resetCoin();

    if (collision(player, coin)) {
        coinSound.play();
        score += 5;

/**
 * @param {integer} init
 * @return { increment: Function, decrement: Function, reset: Function }
 */
var createCounter = function(init) {
    let currentValue = init;

    return {
        increment: () => ++currentValue,
        decrement: () => --currentValue,
        reset: () => {
            currentValue = init;
            return currentValue;
        }
    };
};
